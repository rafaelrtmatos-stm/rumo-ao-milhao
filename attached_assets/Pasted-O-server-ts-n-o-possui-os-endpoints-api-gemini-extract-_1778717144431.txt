O server.ts não possui os endpoints /api/gemini/extract-files, /api/gemini/extract-sale e /api/gemini/analyze-map. Adicione os três antes do middleware do Vite, usando a chave process.env.GEMINI_API_KEY:
tsconst GEMINI_KEY = process.env.GEMINI_API_KEY;
const GEMINI_URL = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash:generateContent?key=${GEMINI_KEY}`;

app.post('/api/gemini/extract-sale', async (req, res) => {
  try {
    const { rawText } = req.body;
    const response = await fetch(GEMINI_URL, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        contents: [{ role: 'user', parts: [{ text: `Extraia os dados do texto abaixo e responda SOMENTE em JSON puro, sem markdown, sem explicações, no formato: {"nomeComprador":"","cpf":"","rg":"","nascimento":"","estadoCivil":"","profissao":"","nacionalidade":"","endereco":"","numero":"","bairro":"","cidade":"","estado":"","cep":"","telefone1":"","numeroLote":"","quadra":"","valorLote":null,"valorEntrada":null,"quantidadeParcelas":null,"valorParcela":null,"vendedor":""}. Campos não encontrados: "" ou null.\n\nRegras:\n- Endereço: separar rua do número\n- Telefones com 10-11 dígitos formatar com DDD\n- Padrão "500 50x350": entrada=500, parcelas=50, valorParcela=350, valorLote=entrada+(parcelas×valorParcela)\n- Estado civil: interpretar solteiro/casado/divorciado/viuvo\n- Corrigir erros de digitação\n\nTexto:\n${rawText}` }] }]
      })
    });
    const data = await response.json();
    const text = data.candidates?.[0]?.content?.parts?.[0]?.text ?? '{}';
    const clean = text.replace(/```json|```/g, '').trim();
    res.json(JSON.parse(clean));
  } catch (err) {
    res.status(500).json({ error: String(err) });
  }
});

app.post('/api/gemini/extract-files', async (req, res) => {
  try {
    const { files } = req.body;
    const parts: any[] = files.map((f: any) => ({ inlineData: { mimeType: f.mimeType, data: f.base64 } }));
    parts.push({ text: `Extraia os dados dos documentos e responda SOMENTE em JSON puro, sem markdown, sem explicações, no formato: {"nomeComprador":"","cpf":"","rg":"","nascimento":"","estadoCivil":"","profissao":"","nacionalidade":"","endereco":"","numero":"","bairro":"","cidade":"","estado":"","cep":"","telefone1":"","numeroLote":"","quadra":"","valorLote":null,"valorEntrada":null,"quantidadeParcelas":null,"valorParcela":null,"vendedor":""}. Campos não encontrados: "" ou null.\n\nRegras:\n- Endereço: separar rua do número\n- Telefones com 10-11 dígitos formatar com DDD\n- Padrão "500 50x350": entrada=500, parcelas=50, valorParcela=350, valorLote=entrada+(parcelas×valorParcela)\n- Estado civil: interpretar solteiro/casado/divorciado/viuvo` });
    const response = await fetch(GEMINI_URL, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ contents: [{ role: 'user', parts }] })
    });
    const data = await response.json();
    const text = data.candidates?.[0]?.content?.parts?.[0]?.text ?? '{}';
    const clean = text.replace(/```json|```/g, '').trim();
    res.json(JSON.parse(clean));
  } catch (err) {
    res.status(500).json({ error: String(err) });
  }
});

app.post('/api/gemini/analyze-map', async (req, res) => {
  try {
    const { base64Data, mimeType } = req.body;
    const response = await fetch(GEMINI_URL, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        contents: [{ role: 'user', parts: [
          { inlineData: { mimeType, data: base64Data } },
          { text: 'Analise este mapa de loteamento e extraia as informações de lotes, quadras e ruas disponíveis. Responda em JSON.' }
        ]}]
      })
    });
    const data = await response.json();
    res.json(data.candidates?.[0]?.content?.parts?.[0] ?? {});
  } catch (err) {
    res.status(500).json({ error: String(err) });
  }
});
